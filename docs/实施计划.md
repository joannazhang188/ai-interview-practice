import os
import re
import fitz
from docx import Document

def extract_text_from_pdf(file_path: str) -> str:
    text = ""
    try:
        doc = fitz.open(file_path)
        for page in doc:
            text += page.get_text()
        doc.close()
    except Exception as e:
        return f"[PDF解析失败: {str(e)}]"
    return text.strip()

def extract_text_from_docx(file_path: str) -> str:
    text = ""
    try:
        doc = Document(file_path)
        for para in doc.paragraphs:
            text += para.text + "\n"
        for table in doc.tables:
            for row in table.rows:
                for cell in row.cells:
                    text += cell.text + " "
                text += "\n"
    except Exception as e:
        return f"[Word解析失败: {str(e)}]"
    return text.strip()

def extract_text(file_path: str) -> str:
    ext = os.path.splitext(file_path)[1].lower()
    if ext == ".pdf":
        return extract_text_from_pdf(file_path)
    elif ext in [".docx", ".doc"]:
        return extract_text_from_docx(file_path)
    elif ext in [".png", ".jpg", ".jpeg"]:
        return "[图片文件暂不支持解析，请上传文字版简历]"
    else:
        return "[不支持的文件格式]"

def extract_skills_from_text(text: str) -> list:
    common_skills = [
        "Python", "Java", "JavaScript", "TypeScript", "Go", "Rust", "C++", "C#",
        "React", "Vue", "Angular", "Node.js", "Django", "Flask", "Spring",
        "SQL", "MySQL", "PostgreSQL", "MongoDB", "Redis",
        "AWS", "Azure", "GCP", "Docker", "Kubernetes", "Linux",
        "Git", "Agile", "Scrum",
        "产品设计", "需求分析", "项目管理", "数据分析", "用户研究",
        "Axure", "Figma", "Sketch", "SQL", "Excel", "PPT",
        "机器学习", "深度学习", "NLP", "计算机视觉",
        "沟通能力", "团队协作", "领导力", "问题解决", "逻辑思维"
    ]
    found = []
    text_upper = text.upper()
    for skill in common_skills:
        if skill.upper() in text_upper:
            found.append(skill)
    return list(set(found))[:20]

def extract_education(text: str) -> list:
    edu_keywords = ["本科", "硕士", "博士", "学士", "研究生", "大学", "学院", "毕业"]
    lines = text.split("\n")
    education = []
    for line in lines:
        for kw in edu_keywords:
            if kw in line and len(line) < 100:
                education.append(line.strip())
                break
    return education[:5]

def extract_experience(text: str) -> list:
    exp_patterns = [
        r'(\d{4}[-/]\d{4}).*?(?:公司|企业|机构)',
        r'(?:负责|主导|参与|担任).{5,80}',
    ]
    results = []
    for pattern in exp_patterns:
        matches = re.findall(pattern, text)
        results.extend(matches)
    return results[:10]

def parse_resume(text: str) -> dict:
    return {
        "skills": extract_skills_from_text(text),
        "education": extract_education(text),
        "experience": extract_experience(text),
        "summary": text[:200] + "..." if len(text) > 200 else text
    }

def parse_portfolio(text: str) -> dict:
    return {
        "summary": text[:300] + "..." if len(text) > 300 else text,
        "skills": extract_skills_from_text(text),
        "highlights": extract_experience(text),
        "suggested_roles": []
    }

    company_name: Optional[str] = Form(""),
    job_title: str = Form(...),
    job_text: str = Form(""),
    file: Optional[UploadFile] = File(None)
):
    raw_text = job_text

    if file:
        ext = os.path.splitext(file.filename)[1].lower()

## Phase 4: SOLO AI 分析接入

### Task 17: SOLO AI 分析流程（会话内执行）

当用户在前端发起匹配度分析或面试回答时，系统会：
1. 创建 AI 请求记录到数据库
2. 前端轮询获取结果
3. **我（SOLO）实时分析请求内容并将结果写入数据库**

**我会在以下时刻介入分析：**

**时机 A：匹配度分析**
- 当 `ai_requests` 表有新记录（request_type = "match_analysis"）
- 我读取 JD + 简历 + 作品内容
- 基于关键词重叠度计算 5 子维度分数
- 生成 6 张分析卡片内容
- 判断阈值，写入结果

**时机 B：面试回答**
- 当有新记录（request_type = "interview_answer" / "sim_answer"）
- 我读取对话历史 + 用户回答
- 训练模式：生成即时评分反馈 + 下一个问题
- 模拟模式：生成下一个问题（不反馈）

**时机 C：面试报告**
- 当 `request_type = "interview_report"`
- 我基于对话历史生成完整报告（10维度评分）

**操作方式**：我会定期检查数据库中的 pending 请求（可通过手动触发或轮询机制），读取上下文，分析后将结果写入 `ai_requests.result` 字段，status 设为 "done"。前端轮询获取到结果后自动渲染。
